148      private static Map<String, JButton> initializeButtons(String... strings)
149      {
150          //Loop through strings and create a hashmap of buttons
151          Map<String, JButton> output = new HashMap<>();
152          for (String value : strings)
153          {
154              JButton btn = new JButton(value);
155              btn.setForeground(Color.DARK_GRAY);
156              btn.setFont(new Font("Arial", Font.BOLD, FONT_SIZE));
157  
158              output.put(value, btn);
159          }
160          return output;
161      }
162      public static void initializeWindow()

166          calcWindow.setSize(300,500);
167          calcWindow.setResizable(false);
168          calcWindow.setTitle("Calculator");
169          buildGUI();
170          calcWindow.setVisible(true);
